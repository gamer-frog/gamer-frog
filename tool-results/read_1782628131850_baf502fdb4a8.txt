   127→@app.after_request
   128→def add_no_cache_headers(response):
   129→    """Apply cache policy by path:
   130→    - HTML/API/state: no-cache (always fresh)
   131→    - /static assets (2xx only): long cache (filenames are versioned with ?v=VERSION_TIMESTAMP)
   132→    - /static assets (non-2xx, e.g. 404): no-cache to prevent CDN from caching errors
   133→    """
   134→    path = (request.path or "")
   135→    if path.startswith('/static/') and 200 <= response.status_code < 300:
   136→        response.headers["Cache-Control"] = "public, max-age=31536000, immutable"
   137→        response.headers.pop("Pragma", None)
   138→        response.headers.pop("Expires", None)
   139→    else:
   140→        response.headers["Cache-Control"] = "no-cache, no-store, must-revalidate, max-age=0"
   141→        response.headers["Pragma"] = "no-cache"
   142→        response.headers["Expires"] = "0"
   143→    return response

   154→def load_state():
   155→    """Load state from file.
   156→
   157→    Includes a simple auto-idle mechanism:
   158→    - If the last update is older than ttl_seconds (default 25s)
   159→      and the state is a "working" state, we fall back to idle.
   160→
   161→    This avoids the UI getting stuck at the desk when no new updates arrive.
   162→    """
   163→    state = None
   164→    if os.path.exists(STATE_FILE):
   165→        try:
   166→            with open(STATE_FILE, "r", encoding="utf-8") as f:
   167→                state = json.load(f)
   168→        except Exception:
   169→            state = None
   170→
   171→    if not isinstance(state, dict):
   172→        state = dict(DEFAULT_STATE)
   173→
   174→    # Auto-idle
   175→    try:
   176→        ttl = int(state.get("ttl_seconds", 300))
   177→        updated_at = state.get("updated_at")
   178→        s = state.get("state", "idle")
   179→        if updated_at and s in WORKING_STATES:
   180→            # tolerate both with/without timezone
   181→            dt = datetime.fromisoformat(updated_at.replace("Z", "+00:00"))
   182→            # Use UTC for aware datetimes; local time for naive.
   183→            if dt.tzinfo:
   184→                from datetime import timezone
   185→                age = (datetime.now(timezone.utc) - dt.astimezone(timezone.utc)).total_seconds()
   186→            else:
   187→                age = (datetime.now() - dt).total_seconds()
   188→            if age > ttl:
   189→                state["state"] = "idle"
   190→                state["detail"] = "待命中（自动回到休息区）"
   191→                state["progress"] = 0
   192→                state["updated_at"] = datetime.now().isoformat()
   193→                # persist the auto-idle so every client sees it consistently
   194→                try:
   195→                    save_state(state)
   196→                except Exception:
   197→                    pass
   198→    except Exception:
   199→        pass
   200→
   201→    return state

   204→def get_office_name_from_identity():
   205→    """Read office display name from OpenClaw workspace IDENTITY.md (Name field) -> 'XXX的办公室'."""
   206→    if not os.path.isfile(IDENTITY_FILE):

   208→    try:
   209→        with open(IDENTITY_FILE, "r", encoding="utf-8") as f:
   210→            content = f.read()
   211→        m = re.search(r"-\s*\*\*Name:\*\*\s*(.+)", content)
   212→        if m:
   213→            name = m.group(1).strip().replace("\r", "").split("\n")[0].strip()
   214→            return f"{name}的办公室" if name else None
   215→    except Exception:
   216→        pass
   217→    return None

   382→def _maybe_apply_random_home_favorite():
   383→    """On page open, randomly apply one saved home favorite if available."""
   384→    global _last_home_rotate_at
   385→
   386→    if not AUTO_ROTATE_HOME_ON_PAGE_OPEN:
   387→        return False, "disabled"
   388→
   389→    try:
   390→        now_ts = datetime.now().timestamp()
   391→        if _last_home_rotate_at and (now_ts - _last_home_rotate_at) < AUTO_ROTATE_MIN_INTERVAL_SECONDS:
   392→            return False, "throttled"
   393→
   394→        idx = _load_home_favorites_index()
   395→        items = idx.get("items") or []
   396→        candidates = []
   397→        for it in items:
   398→            rel = (it.get("path") or "").strip()
   399→            if not rel:
   400→                continue
   401→            abs_path = os.path.join(ROOT_DIR, rel)
   402→            if os.path.exists(abs_path):
   403→                candidates.append((rel, abs_path))
   404→
   405→        if not candidates:
   406→            return False, "no-favorites"
   407→
   408→        rel, src = random.choice(candidates)
   409→        target = FRONTEND_PATH / "office_bg_small.webp"
   410→        if not target.exists():
   411→            return False, "missing-office-bg"
   412→
   413→        shutil.copy2(src, str(target))
   414→        _last_home_rotate_at = now_ts
   415→        return True, rel
   416→    except Exception as e:
   417→        return False, str(e)

   436→def _probe_animated_frame_size(upload_path: str):
   437→    """Return (w,h) from first frame if possible."""
   438→    if Image is not None:
   439→        try:
   440→            with Image.open(upload_path) as im:
   441→                w, h = im.size
   442→                return int(w), int(h)
   443→        except Exception:
   444→            pass
   445→    # ffprobe fallback
   446→    if shutil.which("ffprobe"):
   447→        try:
   448→            cmd = [
   449→                "ffprobe", "-v", "error",
   450→                "-select_streams", "v:0",
   451→                "-show_entries", "stream=width,height",
   452→                "-of", "csv=p=0:s=x",
   453→                upload_path,
   454→            ]
   455→            out = subprocess.check_output(cmd, stderr=subprocess.STDOUT, timeout=5).decode().strip()
   456→            if "x" in out:
   457→                w, h = out.split("x", 1)
   458→                return int(w), int(h)
   459→        except Exception:
   460→            pass
   461→    return None, None

   464→def _animated_to_spritesheet(
   465→    upload_path: str,
   466→    frame_w: int,
   467→    frame_h: int,
   468→    out_ext: str = ".webp",
   469→    preserve_original: bool = True,
   470→    pixel_art: bool = True,
   471→    cols: int | None = None,
   472→    rows: int | None = None,
   473→):
   474→    """Convert animated GIF/WEBP to spritesheet, return (out_path, columns, rows, frames, out_frame_w, out_frame_h)."""
   475→    backend = _ensure_magick_or_ffmpeg_available()
   476→    if not backend:
   477→        raise RuntimeError("未检测到 ImageMagick/ffmpeg，无法自动转换动图")
   478→
   479→    ext = (out_ext or ".webp").lower()
   480→    if ext not in {".webp", ".png"}:
   481→        ext = ".webp"
   482→
   483→    out_fd, out_path = tempfile.mkstemp(suffix=ext)
   484→    os.close(out_fd)
   485→
   486→    with tempfile.TemporaryDirectory() as td:
   487→        frames = 0
   488→        out_fw, out_fh = int(frame_w), int(frame_h)
   489→        if Image is not None:
   490→            try:
   491→                with Image.open(upload_path) as im:
   492→                    n = getattr(im, "n_frames", 1)
   493→                    # 默认保留用户原始帧尺寸（避免先压缩再放大导致像素糊）
   494→                    if preserve_original:
   495→                        out_fw, out_fh = im.size
   496→                    for i in range(n):
   497→                        im.seek(i)
   498→                        fr = im.convert("RGBA")
   499→                        if not preserve_original and (fr.size != (out_fw, out_fh)):
   500→                            resample = Image.Resampling.NEAREST if pixel_art else Image.Resampling.LANCZOS
   501→                            fr = fr.resize((out_fw, out_fh), resample)
   502→                        fr.save(os.path.join(td, f"f_{i:04d}.png"), "PNG")
   503→                    frames = n
   504→            except Exception:
   505→                frames = 0
   506→
   507→        if frames <= 0:
   508→            cmd1 = f"ffmpeg -y -i '{upload_path}' '{td}/f_%04d.png' >/dev/null 2>&1"
   509→            if os.system(cmd1) != 0:
   510→                raise RuntimeError("动图抽帧失败（Pillow/ffmpeg 都失败）")
   511→            files = sorted([x for x in os.listdir(td) if x.startswith("f_") and x.endswith(".png")])
   512→            frames = len(files)
   513→            if frames <= 0:

   563→def normalize_agent_state(s):
   564→    """Normalize agent state for compatibility.
   565→    Maps synonyms (e.g. working/busy -> writing, run/running -> executing) into VALID_AGENT_STATES.

   573→    if s_lower in {'run', 'running', 'execute', 'exec'}:

   575→    if s_lower in {'sync'}:

   580→        return s_lower
   581→    return 'idle'
